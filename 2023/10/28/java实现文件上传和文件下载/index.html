文件上传-base64

private String basepath ="D:\\fileandpicture\\";

@PostMapping("/upload")
// 现在这个message的msg里面存储了一个base64的数据，它是用来存储文件的
public R<String> upload(@RequestBody Message message) throws IOException {
	//获取它的文件名称
    String curpath = basepath + message.getFileName();
    //获取它的base64实际的内容， base64逗号之前的是文件的格式,后面才是真正的内容
    File file = new File(curpath);
    if(file.exists()) file.mkdir();
    FileOutputStream stream = new FileOutputStream(file);
    String base64String = message.getMsg();
    String []parts = base64String.split(",");
    String base64Data = parts[1];
    // 转换为字节数组
    byte[] data = Base64.getDecoder().decode(base64Data);
    // 写入到 本地中（然后存入把它的名字存入到数据库中）
    stream.write(data);
    
    
    
    return xxx;
}

文件下载 - 点击文件之后可下载

后端 - java

@GetMapping("/downloadfile")
public String downloadFile(HttpServletResponse response, HttpServletRequest request, String fileName, String extend) throws IOException {
    JSONObject result = new JSONObject();
    File file = new File(basepath + fileName);
    if (!file.exists()) {
        result.put("error", "下载的文件不存在");
        return result.toString();
    }
    response.reset();
    response.setContentType(extend);
    response.setCharacterEncoding("utf-8");
    response.setContentLength((int) file.length());
    //这个是为了防止中文名称的问题
    response.setHeader("Content-Disposition", "attachment;filename=" + URLEncoder.encode(file.getName(), "UTF-8"));
    byte[] readeBytes = FileUtil.readBytes(file);
    // 写入到request 的 输出流中
    OutputStream os = response.getOutputStream();
    os.write(readeBytes);
    result.put("success", "下载成功");
    return result.toString();
}
